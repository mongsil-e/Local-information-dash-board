
// Additional Security Headers Middleware
app.use((req, res, next) => {
  res.setHeader("X-Content-Type-Options", "nosniff");
  res.setHeader("X-Frame-Options", "SAMEORIGIN"); // Or "DENY" if no framing is ever needed
  // Consider adding Referrer-Policy: strict-origin-when-cross-origin or same-origin if not already handled
  // res.setHeader("Referrer-Policy", "strict-origin-when-cross-origin");
  next();
});
